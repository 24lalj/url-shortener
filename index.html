async function process() { const url = document.getElementById('urlInput').value; const resArea = document.getElementById('resultArea'); const statusMsg = document.getElementById('statusMsg'); const finalLink = document.getElementById('finalLink'); if(!url) return alert("Please enter a URL first!"); try { // CHANGED: We use a relative path '/shorten' instead of 'http://localhost:8000/shorten' // This makes the browser talk to your Railway domain automatically const response = await fetch(`/shorten?long_url=${encodeURIComponent(url)}`, { method: 'POST' }); const data = await response.json(); resArea.style.display = 'block'; finalLink.href = data.short_url; finalLink.innerText = data.short_url; if(data.is_spam) { resArea.className = 'spam-alert'; statusMsg.innerHTML = "⚠️ Spam Warning: This URL looks suspicious:"; } else { resArea.className = 'safe-alert'; statusMsg.innerHTML = "✅ Success: Smart URL generated successfully:"; } } catch (e) { // This error appears if the frontend can't reach the backend alert("Connection error. Ensure your Railway service is 'Online'."); } }
